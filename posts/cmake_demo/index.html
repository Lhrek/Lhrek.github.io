├── include
│   └── Hello.h
└── src
    ├── Hello.cpp
    └── main.cpp

cmake_minimum_required(VERSION 3.5)

project (hello_headers)

# Create a sources variable with a link to all cpp files to compile
set(SOURCES
    src/Hello.cpp
    src/main.cpp
)

# Add an executable with the above sources
add_executable(hello_headers ${SOURCES})

# Set the directories that should be included in the build command for this target
# when running g++ these will be included as -I/directory/path/
target_include_directories(hello_headers 
    PRIVATE 
        ${PROJECT_SOURCE_DIR}/include
)

​ 将生成的库复制到指定位置

    ├── Hello.cpp

cmake_minimum_required(VERSION 3.5)

project(cmake_examples_install)

############################################################
# Create a library
############################################################

#Generate the shared library from the library sources
add_library(cmake_examples_inst SHARED
    src/Hello.cpp
)

target_include_directories(cmake_examples_inst
    PUBLIC 
        ${PROJECT_SOURCE_DIR}/include
)

############################################################
# Create an executable
############################################################

# Add an executable with the above sources
add_executable(cmake_examples_inst_bin
    src/main.cpp
)

# link the new hello_library target with the hello_binary target
target_link_libraries( cmake_examples_inst_bin
    PRIVATE 
        cmake_examples_inst
)

############################################################
# Install
############################################################

# Binaries
install (TARGETS cmake_examples_inst_bin
    DESTINATION bin)

# Library
# Note: may not work on windows
install (TARGETS cmake_examples_inst
    LIBRARY DESTINATION lib)

# Header files
install(DIRECTORY ${PROJECT_SOURCE_DIR}/include/ 
    DESTINATION include)

# Config
install (FILES cmake-examples.conf
    DESTINATION etc)

F-Build-Type

CMAKE_BUILD_TYPE
  1. Release:
    • 为产出的目标进行优化。
    • 不包含调试信息。
    • 通常用于生产或最终发布的版本。
    • 对应的编译器标志（以 GCC 和 Clang 为例）通常包括 -O3。
  2. Debug:
    • 不进行优化。
    • 包含调试信息。
    • 通常用于开发阶段，以便开发者能够使用调试器调试代码。
    • 对应的编译器标志（以 GCC 和 Clang 为例）通常包括 -g。
  3. RelWithDebInfo (Release with Debug Info):
    • 进行优化。
    • 包含调试信息。
    • 在需要优化的同时保留调试信息的情况下使用，例如性能分析。
    • 对应的编译器标志（以 GCC 和 Clang 为例）通常包括 -O2 -g。
  4. MinSizeRel (Minimum Size Release):
    • 优化为最小的产出大小。
    • 不包含调试信息。
    • 在需要优化构建目标大小的应用中使用。
    • 对应的编译器标志（以 GCC 和 Clang 为例）通常包括 -Os。

G-complile-flags

.
├── CMakeLists.txt
├── main.cpp

cmake_minimum_required(VERSION 3.5)

# Set a default C++ compile flag
set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -DEX2" CACHE STRING "Set C++ Compiler Flags" FORCE)

# Set the project name
project (compile_flags)

# Add an executable
add_executable(cmake_examples_compile_flags main.cpp)

target_compile_definitions(cmake_examples_compile_flags 
    PRIVATE EX3
)
  • CMAKE_CXX_FLAGS:
    • 这是一个 CMake 变量，用于存储 C++ 编译器的标志。
  • "${CMAKE_CXX_FLAGS} -DEX2":
    • 这部分代码将 -DEX2 添加到现有的 CMAKE_CXX_FLAGS 变量的内容中。这实际上是为编译器添加了一个定义，名为 EX2。
  • CACHE STRING:
    • 这指示 CMake 将 CMAKE_CXX_FLAGS 变量存储在其缓存中，这样其他 CMake 运行也可以看到和使用它。STRING 指定了该缓存条目的类型，即字符串。
  • "Set C++ Compiler Flags":
    • 这是为缓存条目提供的描述，这样当你使用 CMake GUI 或其他前端查看缓存时，可以看到有关此条目的简短描述。
  • FORCE:
    • 这个选项会强制设置变量的值，即使它已经在缓存中有值了。简而言之，如果 CMAKE_CXX_FLAGS 已经在缓存中有一个值，使用 FORCE 会覆盖这个值。

总的来说，第二行命令的目的是为 C++ 编译器添加一个定义 -DEX2，并确保这个定义在 CMake 的后续运行中也可用（因为它存储在缓存中）。

target_compile_definitions(cmake_examples_compile_flags PRIVATE EX3):

  • 这给特定的目标（在本例中为 cmake_examples_compile_flags）添加了一个编译时定义 EX3。定义只在这个目标的编译中可用，而不影响其他目标。使用 PRIVATE 说明这个定义只应用于这个目标，而不传递给其他链接到它的目标。

H-third-party-library

cmake_minimum_required(VERSION 3.5)

# Set the project name
project (third_party_include)


# find a boost install with the libraries filesystem and system
find_package(Boost 1.46.1 REQUIRED COMPONENTS filesystem system)

# check if boost was found
if(Boost_FOUND)
    message ("boost found")
else()
    message (FATAL_ERROR "Cannot find Boost")
endif()

# Add an executable
add_executable(third_party_include main.cpp)

# link against the boost libraries
target_link_libraries( third_party_include
    PRIVATE
        Boost::filesystem
)

几乎所有重要的项目都需要包含第三方库、标头或程序。 CMake 支持使用 find_package() 函数查找这些工具的路径。这将从 CMAKE_MODULE_PATH 中的文件夹列表中搜索“FindXXX.cmake”格式的 CMake 模块。在 Linux 上，默认搜索路径将包含 /usr/share/cmake/Modules 。

大多数包含的软件包都会设置一个变量 XXX_FOUND ，该变量可用于检查该软件包在系统上是否可用。

I-compiling-with-clang

#!/bin/bash
# Ubuntu supports multiple versions of clang to be installed at the same time.
# The tests need to determine the clang binary before calling cmake
clang_bin=`which clang`
clang_xx_bin=`which clang++`

if [ -z $clang_bin ]; then
    clang_ver=`dpkg --get-selections | grep clang | grep -v -m1 libclang | cut -f1 | cut -d '-' -f2`
    clang_bin="clang-$clang_ver"
    clang_xx_bin="clang++-$clang_ver"
fi

echo "Will use clang [$clang_bin] and clang++ [$clang_xx_bin]"


mkdir -p build.clang && cd build.clang && \
    cmake .. -DCMAKE_C_COMPILER=$clang_bin -DCMAKE_CXX_COMPILER=$clang_xx_bin && make

L-cpp-standard

# Set the minimum version of CMake that can be used
# To find the cmake version run
# $ cmake --version
cmake_minimum_required(VERSION 2.8)

# Set the project name
project (hello_cpp11)

# try conditional compilation
include(CheckCXXCompilerFlag)
CHECK_CXX_COMPILER_FLAG("-std=c++11" COMPILER_SUPPORTS_CXX11)
CHECK_CXX_COMPILER_FLAG("-std=c++0x" COMPILER_SUPPORTS_CXX0X)

# check results and add flag
if(COMPILER_SUPPORTS_CXX11)#
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11")
elseif(COMPILER_SUPPORTS_CXX0X)#
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++0x")
else()
    message(STATUS "The compiler ${CMAKE_CXX_COMPILER} has no C++11 support. Please use a different C++ compiler.")
endif()

# Add an executable
add_executable(hello_cpp11 main.cpp)

# Set the minimum version of CMake that can be used
# To find the cmake version run
# $ cmake --version
cmake_minimum_required(VERSION 3.1)

# Set the project name
project (hello_cpp11)

# set the C++ standard to C++ 11
set(CMAKE_CXX_STANDARD 11)

# Add an executable
add_executable(hello_cpp11 main.cpp)

# Set the minimum version of CMake that can be used
# To find the cmake version run
# $ cmake --version
cmake_minimum_required(VERSION 3.1)

# Set the project name
project (hello_cpp11)

# Add an executable
add_executable(hello_cpp11 main.cpp)

# set the C++ standard to the appropriate standard for using auto
target_compile_features(hello_cpp11 PUBLIC cxx_auto_type)

# Print the list of known compile features for this version of CMake
message("List of compile features: ${CMAKE_CXX_COMPILE_FEATURES}")
